<h3>Examples</h3>

<h4>Usage in jafar form</h4>

```javascript
const Form = require('@jafar-org/react-form/Form').default;
const Field = require('@jafar-org/react-form/Field').default;

const model = {
  id: 'simple',
  fields: {
    canEdit: {
      label: 'Can Edit',
      path: 'canEdit',
      component: {
        name: 'Boolean',
      }
    },
  },
  data: {
      canEdit: true,
  },
};

const resources = {
  components: { 
    Boolean: {
        renderer: Boolean,
    }, 
  }
};

<Form model={model} resources={resources}>
    <Field id='canEdit' />
</Form>
```
